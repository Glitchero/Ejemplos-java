package taller.java;

public class Vehiculo {
public int velocidad;

public int parar() {
velocidad = 0;
return velocidad;

}
public int acelerar(int kmh) {
velocidad += kmh;
return velocidad;
}

}

package taller.java;

public class Coche extends Vehiculo{

	public int gasolina;
	public int recargar(int litros) {
	gasolina+=litros;
	return gasolina;
	}
}
